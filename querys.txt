prefix poster: <http://uib.no/info310/movieOntology#>
prefix owl: <http://www.w3.org/2002/07/owl#>
prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#>
prefix dbpedia: <http://dbpedia.org/resource/>
prefix dbpediaOntology: <http://dbpedia.org/ontology/>
prefix foaf: <http://xmlns.com/foaf/0.1/>

SELECT ?name ?url ?type ?actors
WHERE {
  ?poster a poster:poster;
          poster:posterUrl ?url;
          poster:title ?name;
          poster:title "Bachelorette" .
  
   optional {
    SERVICE <http://dbpedia.org/sparql/> {
      ?x owl:sameAs <http://yago-knowledge.org/resource/Pearl_Harbour_(film)>;
         dbpediaOntology:abstract ?type;
         dbpediaOntology:starring ?actors .
    }
    filter (langMatches(lang(?type), "EN"))
  }
  
}

limit 20
